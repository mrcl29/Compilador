/// Aquest programa té molts de càlculs una mica aleatoris i confosos,
/// simplement es per mostrar com funciona la entrada i sortida, la crida de funcions,
/// la declaració de variables amb asignacions diferents, la sintaxis general...

/// Funció que retorna un càlcul de dos valors segons si li pases true o false
fnct integer calcul(x::integer, y::logical, z::integer):
    val t::integer = 0;
    if(y /= false):
        t = (x + z - 10);
    else:
        t = (x + z + 10);
    endif
    rtrn (t);
endfnct

/// Declaració de variables i constants
con xx::integer = 100;
val zz::integer = 10;
con xz1234::logical = (TRUE);

/// Un bucle que modifica una variable usada, i imprimeix la iteració del bucle i el resultat de la funció
for index::integer = 1 to 10:
    out(index);
    out(calcul(xx,xz1234,zz));
endfor
