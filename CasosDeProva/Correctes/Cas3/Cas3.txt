
/// Aqui vorem com funcionen les tuples i provarem un codi amb moltes iteracions.
/// També serveix per veure com el codi optimitzat es un poc més ràpid que l'original.
/// Es veu com es pot fer servir la variable creada al bucle for com una variable global.

/// Si li dones el valor true a 'y' llavors la penúltima linia ha de ser true, sino no s'imprimeix. La última linia ha de ser sempre true.
/// També s'ha de veure la impresió de totes les iteracions del bucle.

tuple x = {integer, logical};
x[0] = 10000;

con y::logical = in(logical);

for z::integer = 0 to x[0]:
    out(z);
    if(y === TRUE && z === (x[0] - 1)):
        x[1] = y;
        out(x[1]);
    endif
endfor

x[1] = true;
x[1] = ((x[1] || y) && z === x[0]);
out(x[1]);
